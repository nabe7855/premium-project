セットアップ全体像

もりさんの環境（Search Console + GA4 + Vercel SEO + X未運用 + 独自開発サイト）に最適化した設定手順です。

P1 Google Search Console API → GAS 検索順位・クリック数の自動取得 無料 30分
P1 GA4 Data API → GAS PV・ユーザー数・流入元の自動取得 無料 30分
P2 X (Twitter) API フォロワー数・投稿分析の自動取得 無料(Basic) / $100/月(Basic+) 1時間

Google Search Console API → GAS 設定

検索クエリ別の表示回数・クリック数・平均順位を毎日自動取得し、Google Sheetsに蓄積します。

1

Search Consoleでサイトを確認

Search Consoleにアクセスし、ストロベリーボーイズのサイトが「プロパティ」として登録されていることを確認してください。未登録なら「プロパティを追加」から登録します。

2

Google Sheetsを新規作成

「SEO自動レポート_ストロベリーボーイズ」という名前で新しいスプレッドシートを作成します。シート名を「SearchConsole」にリネームしてください。

3

Apps Script を開く

スプレッドシートの「拡張機能」→「Apps Script」を開きます。

4

以下のコードを貼り付ける

// ===== Google Search Console 自動データ取得 ===== // ストロベリーボーイズ SEO運用向け const SITE_URL = 'https://your-site.com/'; // ← サイトURLに変更 const GSC_SHEET_NAME = 'SearchConsole'; // ターゲットKW（順位を特に追跡したいもの） const TARGET_KEYWORDS = [ '女性用風俗 横浜', '女性用風俗 福岡', '女性用風俗 神奈川', '女性用風俗 博多', '女性向け 風俗 横浜', '女性向け 風俗 福岡', 'レディースコミック 横浜', 'ストロベリーボーイズ', ]; function fetchSearchConsoleData() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(GSC_SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(GSC_SHEET_NAME); } // ヘッダーが無ければ追加 if (sheet.getLastRow() === 0) { sheet.appendRow([ '取得日', '対象日', 'クエリ', 'クリック数', '表示回数', 'CTR', '平均順位', 'ターゲットKW' ]); // ヘッダー書式設定 sheet.getRange(1, 1, 1, 8).setFontWeight('bold') .setBackground('#2B5797').setFontColor('#FFFFFF'); } // 過去3日分のデータを取得（当日は未確定のため除外） const today = new Date(); const endDate = new Date(today.getTime() - 1 * 86400000); const startDate = new Date(today.getTime() - 3 * 86400000); const request = { startDate: formatDate(startDate), endDate: formatDate(endDate), dimensions: ['query', 'date'], rowLimit: 1000, // startRow: 0 (デフォルト) }; try { const response = SearchConsole.Searchanalytics .query(request, SITE_URL); if (!response.rows || response.rows.length === 0) { Logger.log('データなし'); return; } const fetchDate = formatDate(today); const rows = response.rows.map(row => { const query = row.keys[0]; const date = row.keys[1]; const isTarget = TARGET_KEYWORDS .some(kw => query.includes(kw)) ? 'YES' : ''; return [ fetchDate, date, query, row.clicks, row.impressions, (row.ctr * 100).toFixed(2) + '%', row.position.toFixed(1), isTarget ]; }); // 一括書き込み（高速） const lastRow = sheet.getLastRow(); sheet.getRange(lastRow + 1, 1, rows.length, 8) .setValues(rows); Logger.log(`${rows.length}行追加完了`); } catch (e) { Logger.log('エラー: ' + e.message); } } function formatDate(d) { return Utilities.formatDate(d, 'Asia/Tokyo', 'yyyy-MM-dd'); } // ===== トリガー設定用 ===== function createDailyTrigger_GSC() { ScriptApp.newTrigger('fetchSearchConsoleData') .timeBased() .atHour(6) // 毎朝6時に実行 .everyDays(1) .create(); }
5

Search Console APIを有効化

Apps Scriptエディタの左メニュー「サービス」→「+」→「Google Search Console API」を検索して追加します。識別子は「SearchConsole」のままにしてください。

6

SITE_URLを自分のサイトに変更

コード1行目の 'https://your-site.com/' をストロベリーボーイズのサイトURLに変更します。Search Consoleに登録した形式と完全一致させてください（末尾スラッシュの有無も）。

7

初回実行＆トリガー設定

まず fetchSearchConsoleData を手動実行して権限承認。次に createDailyTrigger_GSC を実行して毎朝6時の自動実行を設定します。

注意: 初回実行時にGoogleの権限承認画面が出ます。「詳細」→「（安全でない）に移動」→ 許可してください。自分のGASなので安全です。

GA4 Data API → GAS 設定

日別のPV・ユーザー数・セッション数・流入元を自動取得します。

1

GA4のプロパティIDを確認

GA4 → 管理 → プロパティ設定 → 「プロパティID」の数字をメモします（例: 123456789）。

2

同じスプレッドシートに「GA4」シートを追加

Search Consoleと同じスプレッドシートに、新しいシート「GA4」を追加します。

3

以下のコードをApps Scriptに追加

// ===== GA4 Data API 自動データ取得 ===== const GA4_PROPERTY_ID = '123456789'; // ← GA4プロパティIDに変更 const GA4_SHEET_NAME = 'GA4'; function fetchGA4Data() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(GA4_SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(GA4_SHEET_NAME); } if (sheet.getLastRow() === 0) { sheet.appendRow([ '取得日', '対象日', 'アクティブユーザー', 'セッション数', 'PV数', '平均セッション時間(秒)', '新規ユーザー率' ]); sheet.getRange(1, 1, 1, 7).setFontWeight('bold') .setBackground('#2B5797').setFontColor('#FFFFFF'); } const today = new Date(); const endDate = new Date(today.getTime() - 86400000); const startDate = new Date(today.getTime() - 3 * 86400000); const request = AnalyticsData.newRunReportRequest(); request.dateRanges = [AnalyticsData.newDateRange()]; request.dateRanges[0].startDate = formatDate(startDate); request.dateRanges[0].endDate = formatDate(endDate); request.dimensions = [ { name: 'date' }, ]; request.metrics = [ { name: 'activeUsers' }, { name: 'sessions' }, { name: 'screenPageViews' }, { name: 'averageSessionDuration' }, { name: 'newUsers' }, ]; try { const response = AnalyticsData.Properties .runReport(request, `properties/${GA4_PROPERTY_ID}`); if (!response.rows) { Logger.log('GA4データなし'); return; } const fetchDate = formatDate(today); const rows = response.rows.map(row => { const d = row.dimensionValues[0].value; const dateStr = d.substring(0,4)+'-'+d.substring(4,6)+'-'+d.substring(6,8); const m = row.metricValues; const activeUsers = parseInt(m[0].value); const newUsers = parseInt(m[4].value); const newRate = activeUsers > 0 ? ((newUsers / activeUsers) * 100).toFixed(1) + '%' : '0%'; return [ fetchDate, dateStr, activeUsers, parseInt(m[1].value), parseInt(m[2].value), parseFloat(m[3].value).toFixed(1), newRate ]; }); const lastRow = sheet.getLastRow(); sheet.getRange(lastRow + 1, 1, rows.length, 7) .setValues(rows); Logger.log(`GA4: ${rows.length}行追加`); } catch (e) { Logger.log('GA4エラー: ' + e.message); } } // ===== 流入元別レポート ===== function fetchGA4TrafficSources() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName('GA4_流入元'); if (!sheet) { sheet = ss.insertSheet('GA4_流入元'); sheet.appendRow([ '取得日', 'チャネル', 'ソース/メディア', 'セッション数', 'ユーザー数' ]); sheet.getRange(1, 1, 1, 5).setFontWeight('bold') .setBackground('#2B5797').setFontColor('#FFFFFF'); } const today = new Date(); const endDate = new Date(today.getTime() - 86400000); const startDate = new Date(today.getTime() - 7 * 86400000); const request = AnalyticsData.newRunReportRequest(); request.dateRanges = [AnalyticsData.newDateRange()]; request.dateRanges[0].startDate = formatDate(startDate); request.dateRanges[0].endDate = formatDate(endDate); request.dimensions = [ { name: 'sessionDefaultChannelGroup' }, { name: 'sessionSourceMedium' }, ]; request.metrics = [ { name: 'sessions' }, { name: 'activeUsers' }, ]; request.orderBys = [{ metric: { metricName: 'sessions' }, desc: true }]; request.limit = 50; try { const response = AnalyticsData.Properties .runReport(request, `properties/${GA4_PROPERTY_ID}`); if (!response.rows) return; const fetchDate = formatDate(today); const rows = response.rows.map(r => [ fetchDate, r.dimensionValues[0].value, r.dimensionValues[1].value, parseInt(r.metricValues[0].value), parseInt(r.metricValues[1].value), ]); const lastRow = sheet.getLastRow(); sheet.getRange(lastRow + 1, 1, rows.length, 5) .setValues(rows); } catch(e) { Logger.log('流入元エラー: ' + e.message); } } // ===== 全APIまとめて実行（トリガー用） ===== function dailyDataCollection() { fetchSearchConsoleData(); fetchGA4Data(); fetchGA4TrafficSources(); Logger.log('全データ取得完了: ' + new Date()); } // これを1回実行してトリガー設定 function createDailyTrigger_All() { // 既存トリガーを削除 ScriptApp.getProjectTriggers().forEach(t => { if (t.getHandlerFunction() === 'dailyDataCollection') { ScriptApp.deleteTrigger(t); } }); // 毎朝6時に全データ取得 ScriptApp.newTrigger('dailyDataCollection') .timeBased() .atHour(6) .everyDays(1) .create(); Logger.log('毎朝6時トリガー設定完了'); }
4

GA4 Data APIを有効化

Apps Scriptの「サービス」→「+」→「Google Analytics Data API」を追加。識別子は「AnalyticsData」にしてください。

5

GA4_PROPERTY_IDを変更して実行

プロパティIDを自分のものに変更し、まず dailyDataCollection を手動実行。成功したら createDailyTrigger_All を実行してトリガー設定完了です。

一括トリガー: createDailyTrigger_All を実行すれば、Search Console + GA4 + 流入元の3つのデータが毎朝6時にまとめて取得されます。Coworkスケジュールタスク（7時）の前に完了するので、投稿生成時にデータが参照できます。

Vercel Analytics 連携

Vercelで独自開発サイトをホスティングしている場合、Web VitalsやアクセスデータをAPI経由で取得できます。

1

Vercel APIトークンを取得

Vercel Settings → Tokens で新しいトークンを生成します。名前は「SEO-Analytics」、スコープは「Read Only」で十分です。

2

プロジェクトIDを確認

Vercelダッシュボード → プロジェクト → Settings → General → 「Project ID」をメモします。

3

GASにVercelデータ取得を追加

// ===== Vercel Analytics データ取得 ===== const VERCEL_TOKEN = 'your-vercel-token'; // ← トークンに変更 const VERCEL_PROJECT_ID = 'your-project-id'; // ← プロジェクトIDに変更 function fetchVercelAnalytics() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName('Vercel'); if (!sheet) { sheet = ss.insertSheet('Vercel'); sheet.appendRow([ '取得日', 'PV(24h)', 'ユニークビジター', 'トップページ', 'トップリファラー' ]); sheet.getRange(1, 1, 1, 5).setFontWeight('bold') .setBackground('#2B5797').setFontColor('#FFFFFF'); } const headers = { 'Authorization': `Bearer ${VERCEL_TOKEN}`, }; // Web Analytics の概要データ取得 const now = Date.now(); const from = now - 86400000; // 24時間前 const url = `https://vercel.com/api/web/insights/stats` + `?projectId=${VERCEL_PROJECT_ID}` + `&from=${from}&to=${now}` + `&environment=production`; try { const res = UrlFetchApp.fetch(url, { headers: headers, muteHttpExceptions: true }); if (res.getResponseCode() !== 200) { Logger.log('Vercel API Error: ' + res.getContentText()); return; } const data = JSON.parse(res.getContentText()); const fetchDate = formatDate(new Date()); sheet.appendRow([ fetchDate, data.pageViews || 0, data.visitors || 0, data.topPages?.[0]?.page || '-', data.topReferrers?.[0]?.referrer || '-', ]); Logger.log('Vercelデータ取得完了'); } catch(e) { Logger.log('Vercelエラー: ' + e.message); } }
Vercel APIの注意: Web AnalyticsはVercelのProプラン以上で利用可能です。Hobbyプランの場合、Vercel Speed Insights（Web Vitals）のみ取得可能です。Analytics機能の有効状態をVercelダッシュボードで確認してください。

X (Twitter) API 設定

フォロワー数の推移と投稿パフォーマンスを自動取得します。

1

X Developer Portalに登録

X Developer Portalにアクセスし、開発者アカウントを申請します。個人利用・ビジネスプロモーション用途で申請してください。

2

プランを選択

Free: 月1,500ツイート書き込み、読み取り制限あり。フォロワー数取得には十分。
Basic ($100/月): 月10,000読み取り、投稿分析に必要。本格運用にはこちらが必要です。
まずはFreeプランで始めて、必要に応じてアップグレードするのがおすすめです。

3

APIキーを取得

Developer Portal → Projects & Apps → Keys and Tokens で以下を取得します:

API Key, API Secret Key, Bearer Token, Access Token, Access Token Secret

4

GASにX データ取得を追加

// ===== X (Twitter) API データ取得 ===== // Free プランでも動くフォロワー数取得 const X_BEARER_TOKEN = 'your-bearer-token'; // ← Bearer Tokenに変更 const X_USER_ID = 'your-user-id'; // ← XのユーザーIDに変更 // ユーザーIDの調べ方: https://tweeterid.com/ にユーザー名を入力 function fetchXFollowerCount() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName('X_Analytics'); if (!sheet) { sheet = ss.insertSheet('X_Analytics'); sheet.appendRow([ '取得日', 'フォロワー数', 'フォロー数', 'ツイート数', '前日比フォロワー' ]); sheet.getRange(1, 1, 1, 5).setFontWeight('bold') .setBackground('#2B5797').setFontColor('#FFFFFF'); } const url = `https://api.x.com/2/users/${X_USER_ID}` + '?user.fields=public_metrics'; try { const res = UrlFetchApp.fetch(url, { headers: { 'Authorization': `Bearer ${X_BEARER_TOKEN}` }, muteHttpExceptions: true }); if (res.getResponseCode() !== 200) { Logger.log('X API Error: ' + res.getContentText()); return; } const data = JSON.parse(res.getContentText()); const metrics = data.data.public_metrics; const followers = metrics.followers_count; const following = metrics.following_count; const tweets = metrics.tweet_count; // 前日比計算 const lastRow = sheet.getLastRow(); let diff = '-'; if (lastRow > 1) { const prevFollowers = sheet.getRange(lastRow, 2).getValue(); diff = followers - prevFollowers; if (diff > 0) diff = '+' + diff; } const fetchDate = formatDate(new Date()); sheet.appendRow([fetchDate, followers, following, tweets, diff]); Logger.log(`X: フォロワー ${followers} (${diff})`); } catch(e) { Logger.log('Xエラー: ' + e.message); } } // ===== 投稿パフォーマンス取得（Basic プラン以上） ===== function fetchXRecentTweets() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName('X_PostLog'); if (!sheet) { sheet = ss.insertSheet('X_PostLog'); sheet.appendRow([ '取得日', '投稿日時', 'テキスト', 'インプレッション', 'いいね', 'RT', 'リプライ', 'ER%' ]); sheet.getRange(1, 1, 1, 8).setFontWeight('bold') .setBackground('#2B5797').setFontColor('#FFFFFF'); } const url = `https://api.x.com/2/users/${X_USER_ID}/tweets` + '?max_results=10' + '&tweet.fields=created_at,public_metrics'; try { const res = UrlFetchApp.fetch(url, { headers: { 'Authorization': `Bearer ${X_BEARER_TOKEN}` }, muteHttpExceptions: true }); if (res.getResponseCode() !== 200) { Logger.log('X Tweets Error: ' + res.getContentText()); return; } const data = JSON.parse(res.getContentText()); if (!data.data) return; const fetchDate = formatDate(new Date()); const rows = data.data.map(tweet => { const m = tweet.public_metrics; const imp = m.impression_count || 0; const eng = m.like_count + m.retweet_count + m.reply_count; const er = imp > 0 ? ((eng / imp) * 100).toFixed(2) + '%' : '0%'; return [ fetchDate, tweet.created_at, tweet.text.substring(0, 100), imp, m.like_count, m.retweet_count, m.reply_count, er ]; }); const lastRow = sheet.getLastRow(); sheet.getRange(lastRow + 1, 1, rows.length, 8) .setValues(rows); Logger.log(`X投稿: ${rows.length}件取得`); } catch(e) { Logger.log('X投稿エラー: ' + e.message); } }
Freeプランで始める場合: fetchXFollowerCountだけ使えばOK。毎日のフォロワー数推移が自動で記録されます。投稿パフォーマンス取得(fetchXRecentTweets)にはBasicプラン($100/月)が必要ですが、投稿が増えてきたら検討しましょう。
X APIの注意点: Bearer TokenはApps Script内にハードコードせず、プロパティサービス (PropertiesService.getScriptProperties()) に保存するのがセキュリティ的にベターです。ただし、個人利用のスクリプトなら上記の方法でも問題ありません。
